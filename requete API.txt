SELECT TOP(50000) 
    Id, CreationDate, Score, ViewCount, AnswerCount, CommentCount, FavoriteCount, Title, Body, Tags
FROM 
    Posts
WHERE 
    CreationDate BETWEEN CONVERT(datetime, '2022-01-01') AND CONVERT(datetime, '2022-12-31')
    AND CommentCount IS NOT NULL
    AND FavoriteCount IS NOT NULL
    AND Score > 4
    AND ViewCount > 40
    AND AnswerCount > 3
    AND CommentCount > 3
ORDER BY CreationDate
